        const filteredTools = category === 'all' ? tools : tools.filter(t => t.category === category);

        grid.innerHTML = filteredTools.map(tool => `
            <div class="tool-card p-5" data-category="${tool.category}">
                <div class="flex items-start gap-4">
                    <div class="w-12 h-12 rounded-lg ${colorClasses[tool.color].bg} flex items-center justify-center flex-shrink-0">
                        <i class="fas ${tool.icon} ${colorClasses[tool.color].text}"></i>
                    </div>
                    <div class="flex-1 min-w-0">
                        <div class="flex items-center gap-2 mb-2">
                            <h4 class="font-mono text-sm font-semibold text-[#111827] truncate">${tool.name}</h4>
                        </div>
                        <p class="text-[#475569] text-sm leading-relaxed">${tool.desc}</p>
                    </div>
                </div>
            </div>
        `).join('');
    }

    // Category tabs
    document.querySelectorAll('.category-tab').forEach(tab => {
        tab.addEventListener('click', () => {
            document.querySelectorAll('.category-tab').forEach(t => {
                t.classList.remove('active');
                t.classList.add('bg-white', 'border', 'border-[#E2E8F0]', 'text-[#475569]');
            });
            tab.classList.add('active');
            tab.classList.remove('bg-white', 'border', 'border-[#E2E8F0]', 'text-[#475569]');
            renderTools(tab.dataset.category);
        });
    });

    // Initial render
    renderTools();

    // Smooth scroll for anchor links
    document.querySelectorAll('a[href^="#"]').forEach(anchor => {
        anchor.addEventListener('click', function (e) {
            e.preventDefault();
            const target = document.querySelector(this.getAttribute('href'));
            if (target) {
                target.scrollIntoView({ behavior: 'smooth', block: 'start' });
                mobileMenu.classList.add('hidden');
            }
        });
    });
</script>
